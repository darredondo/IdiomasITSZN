//Funcion para señalar la opcion seleccionada

$(document).on('click', '.opcion', function () {

   $('.spcion').removeClass((active");

   $(this).addClass("active");

});

 

//Opcion de regresar

$(document).on('click', '#BTNregresar', function () {

   window.close();

});

 

$(document).on('click', '.BTNImprimir', function () {

   var isCurso=$(this).attr('idCurso');

   openNewTab('./script/PDFacta.php?idCurso=' + idCurso);

});

 

function openNewTab(url) {

   var win = window.open(url, '_blank');

   if (win) {

       //Browser has allowed it to bewopened

       win.focus();

   } else {

       //Browser has blocked it

       alert('Please allow popups for this website');

   }

}
